        function showPage(pageId) {
            const nextPage = document.getElementById('page-' + pageId);
            if (!nextPage) return;

            // Update Nav links
            document.querySelectorAll('.nav-link').forEach(link => {
                link.classList.remove('active');
                if(link.dataset.page === pageId) link.classList.add('active');
            });

            const currentPage = document.querySelector('.page-content.active');
            if (currentPage && currentPage !== nextPage) {
                gsap.to(currentPage, { 
                    opacity: 0, 
                    y: 20, 
                    duration: 0.3, 
                    onComplete: () => {
                        currentPage.classList.remove('active');
                        activatePage(nextPage, pageId);
                    }
                });
            } else {
                activatePage(nextPage, pageId);
            }
        }

        function activatePage(element, id) {
            element.classList.add('active');
            gsap.fromTo(element, { opacity: 0, y: 20 }, { opacity: 1, y: 0, duration: 0.5 });
            
            if(id === 'home') {
                startStatCounter();
                renderFeaturedProjects();
            }
            if(id === 'projects') renderProjects('all');
            if(id === 'news') renderNews();
            window.scrollTo(0,0);
        }

        function renderProjects(filter) {
            const grid = document.getElementById('project-grid');
            if (!grid) return;
            const filtered = filter === 'all' ? projectsData : projectsData.filter(p => p.category === filter);
            grid.innerHTML = filtered.map(p => createProjectCard(p)).join('');
            gsap.from("#project-grid > div", { opacity: 0, y: 30, duration: 0.5, stagger: 0.1 });
        }

        function startStatCounter() {
            document.querySelectorAll('.stat-number').forEach(stat => {
                const target = parseInt(stat.dataset.target);
                let current = 0;
                const update = () => {
                    const step = target / 50;
                    if(current < target) {
                        current += step;
                        stat.innerText = Math.ceil(current) + (target > 500 ? '+' : '');
                        requestAnimationFrame(update);
                    } else {
                        stat.innerText = target + (target > 500 ? '+' : '');
                    }
                }
                update();
